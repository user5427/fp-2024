<command> ::= <add_bus> | <remove_bus> | <add_train> | <remove_train> | 
<add_bus_stop> | <remove_bus_stop> | <add_train_stop> | <remove_train_stop> | 
<create_route> | <assign_bus_to_route> | <assign_train_to_route> | <remove_bus_from_route> | <remove_train_from_route> |
<add_bus_stop_to_route> | <remove_bus_stop_from_route> | <add_train_stop_to_route> | <remove_train_stop_from_route> |
<print_all_routes> | <print_all_busses> | <print_all_trains> |
<print_all_route_busses> | <print_all_route_trains> |
<join_two_routes> | <join_two_routes_at_stop> | 
<create_path> | <remove_path>

<add_bus> ::= "add_bus" <bus_id> <bus_park> <bus_working_start_time> <bus_working_end_time>
<remove_bus> ::= "remove_bus" <bus_id>
<add_train> ::= "add_train" <train_id> <train_yard> <train_working_start_time> <train_working_end_time>
<remove_train> ::= "remove_train> <train_id>
<add_bus_stop> ::= "add_bus_stop" <stop_id> <name> <coord-x> <coord-y>
<remove_bus_stop> ::= "remove_bus_stop" <stop_id>
<add_train_stop> ::= "add_train_stop" <stop_id> <name> <coord-x> <coord-y>
<remove_train_stop> ::= "remove_train_stop" <stop_id>

<create_route> ::= "create_route" <route_id> <route_name>
<assign_bus_to_route> ::= "assign_bus_to_route" <bus_id> <route_id>
<assign_train_to_route> ::= "assign_train_to_route" <train_id> <route_id>
<remove_bus_from_route> ::= "remove_bus_from_route" <bus_id> <route_id>
<remove_train_from_route> ::= "remove_train_from_route" <train_id> <route_id>
<add_bus_stop_to_route> ::= "add_bus_stop_to_route" <stop_id> <route_id>
<remove_bus_stop_from_route> ::= "remove_bus_stop_from_route" <stop_id> <route_id>
<add_train_stop_to_route> ::= "add_train_stop_to_route" <stop_id> <route_id>
<remove_train_stop_from_route> ::= "remove_train_stop_from_route" <stop_id> <route_id>

<print_all_routes> ::= "print_all_routes"
<print_all_busses> ::= "print_all_busses"
<print_all_trains> ::= "print_all_trains"
<print_all_route_busses> ::= "print_all_route_busses" <route_id>
<print_all_route_trains> ::= "print_all_route_trains" <route_id>

<join_two_routes> ::= "join_two_routes" <route_id> <route_id> <route_id> <route_name>
<join_two_routes_at_stop> ::= "join_two_routes" <route_id> <route_id> <stop_id> <route_id> <route_name>

<create_path> ::= "create_path" <path_id> <name> <path_length> <stop_id> <stop_id>
<remove_path> ::= "remove_path" <path_id>


<bus_id> ::= <integer>
<bus_park> ::= <string>
<bus_working_start_time> ::= <string>
<bus_working_end_time> ::= <string>
<train_id> ::= <integer>
<train_yard> ::= <string>
<train_working_start_time> ::= <string>
<train_working_end_time> ::= <string>
<stop_id> ::= <integer>
<name> ::= <string>
<coord-x> ::= <float>
<coord-y> ::= <float>
<route_id> ::= <integer>
<route_name> ::= <string>
<path_id> ::= <integer>
<path_length> ::= <float>
<float> ::= "[+-]?[0-9]+([.][0-9]+)?"
<string> ::= "[a-zA-Z0-9_ ]+"
<integer> ::= "[0-9]+"
